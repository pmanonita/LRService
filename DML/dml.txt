CREATE TABLE User (
id         INT (11) NOT NULL,
userName   VARCHAR (20) NOT NULL,
password   VARCHAR (20) NOT NULL,
firstName  VARCHAR (20),
lastName   VARCHAR (20),
email      VARCHAR (20),
mobile     BIGINT (20), 
serviceKey VARCHAR (255) NOT NULL,
authKey    VARCHAR (255),
role       VARCHAR (20),
createDate DATE NOT NULL,
PRIMARY KEY ( id )
);
